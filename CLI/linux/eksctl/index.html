    eksctl-cmd¶

    install¶

    # install eksctl
# consider install eksctl version 0.89.0
# if you have older version yaml 
# https://eksctl.io/announcements/nodegroup-override-announcement/
curl --location "https://github.com/weaveworks/eksctl/releases/latest/download/eksctl_$(uname -s)_amd64.tar.gz" | tar xz -C /tmp
sudo mv -v /tmp/eksctl /usr/local/bin
eksctl completion bash >> ~/.bash_completion
. /etc/profile.d/bash_completion.sh
. ~/.bash_completion

    iamidentitymapping¶

    eksctl get iamidentitymapping \
--cluster ekscluster1 \
--region us-east-2

eksctl create iamidentitymapping \
--cluster ekscluster1 \
--region us-east-2 \
--arn arn:aws:iam::xxxx:role/eksadmin \
--group system:masters \
--username admin

    create windows self-managed node group¶

    cluster_name=myeksctl
region_name=ap-southeast-1
eksctl create nodegroup   \
  --region $region_name   \
  --cluster $cluster_name   \
  --name nodegroup-win-1   \
  --node-type m5.xlarge   \
  --nodes 1   \
  --nodes-min 1   \
  --nodes-max 2 \
  --node-ami-family WindowsServer2019FullContainer \
  --managed=false

    cluster_name=myeksctl
region_name=ap-southeast-1
eksctl create nodegroup   \
  --region $region_name   \
  --cluster $cluster_name   \
  --name nodegroup-br-1   \
  --node-type m5.xlarge   \
  --nodes 1   \
  --nodes-min 1   \
  --nodes-max 2 \
  --node-ami-family Bottlerocket

    create nodegroup¶

    cluster_name=ekscluster2
region_name=us-east-1
eksctl create nodegroup   \
  --region $region_name   \
  --cluster $cluster_name   \
  --name mng-a   \
  --node-type m5.large   \
  --nodes 2   \
  --node-private-networking \
  --subnet-ids subnet-aaa,subnet-bbb

    scale nodegroup¶

    CLUSTER_NAME=ekscluster1
NODEGROUP_NAME=managed-ng
AWS_REGION=us-east-2

eksctl scale nodegroup \
--cluster=${CLUSTER_NAME} \
--region ${AWS_REGION} \
--nodes=3 \
${NODEGROUP_NAME}

    appmesh cluster¶

    eksctl create cluster \
--name appmeshtest \
--nodes-min 2 \
--nodes-max 3 \
--nodes 2 \
--auto-kubeconfig \
--full-ecr-access \
--appmesh-access
